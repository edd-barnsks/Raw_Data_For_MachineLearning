interface User {
    id: number;

async function fetchUser(id: number): Promise<User> {
    const res = await fetch(`/api/users/${id}`);
    const data: User = await res.json();

}

const isPrime = (n: number): boolean => {
    if (n < 2) return false;
    for (let i = 2; i <= Math.sqrt(n); i++) {
        if (n % i === 0) return false;
    }
    return true;
};

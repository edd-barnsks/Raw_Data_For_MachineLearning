public bool IsPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= Math.Sqrt(n); i++) {
        if (n % i == 0) return false;
    }
    return true;
}

public int Fibonacci(int n) {
    if (n <= 1) return n;
    return Fibonacci(n - 1) + Fibonacci(n - 2);
}
